# REST API

# Создание транзакции

# Метод: POST

# Endpoint: https://mediacube.id/payments

# Параметры запроса:

# identify:

- обязательный
- string

# purpose:

- обязательный 
- enum(1 - License Fee, 2 - ADVERTISING_PAYMENT)

# request_id:

- обязательный
- string
- случайно сгенерированная уникальная строка
- количество символов: 16-255

# description:

- обязательный
- string
- сообщение о снятии средств с вашего счета MediaCube. Будет отображено во вкладке _Транзакции_ в Mediacube.ID

# data:

- обязательный
- array

# data[*] wallet:

- обязательный
- string
- существующий номер кошелька пользователя системы MediaCube

# data[*] amount:

- обязательный
- numeric

# data[*] description:

- обязательный
- string     
- сообщение о приходе средств. Будет отображено во вкладке _Транзакции_ в Mediacube.ID.

# Пример запроса:

        $headers = [
            'Authorization' => 'Bearer <jwt-token>',
            'Content-Type' => 'application/json', 
            'X-Requested-With' => 'XMLHttpRequest',
            'Accept' => 'application/json',
        ];

        $requestData = [
            'identify' => '<identify-key>',
            'request_id' => 'c02e79a2-d30c-46e3-b3dd-c0e6a987acf8',
            'description' => 'Выплаты по реферальной системе MediaCube Music',
            'data' => [
                [
                    'description' => 'Приход с реферальной системы MediaCube Music',
                    'amount' => '10.423',
                    'wallet' => '1111222233334444',
                ],
                [
                    'description' => 'Приход с реферальной системы MediaCube Music',
                    'amount' => '6.8',
                    'wallet' => '5555666677778888',
                ],
            ],
        ];

        $response = $httpClient->post('https://mediacube.id/payments', [
            'form_params' => $requestData, 
            'headers' => $headers
        ]);

# Возможные ответы:

# 201

{
  "data": {
    "request_id":"c02e79a2-d30c-46e3-b3dd-c0e6a987acf8",
    "status": 2,
    "err_msg": null
   }
}

# Status

1 - processing
2 - success
3 - error

# 401

У вашего Access Token нет доступа

# 403

У вашего Identify Key не хватает прав на совершение операции

# 422

Неверные данные
