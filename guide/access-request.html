# Получение доступа

# Аутентификация

# Пример на PHP:

$response = $httpClient->post('https://mediacube.id/oauth/token', [
    'form_params' => [
        'grant_type' => 'partner',
        'client_id' => '<client-id>',
        'client_secret' => '<client-secret>',
    ],
]);

# Пример на JavaScript:

async function getAuthToken(){
    const body = {
        grant_type: 'partner',
        client_id: '<client-id>',
        client_secret: '<client-secret>'
    }   
    
    const response = await axios.post('https://mediacube.id/oauth/token', body)
    
    return response.access_token
}

# Пример успешного(200) ответа:

{
  "token_type": "Bearer",
  "expires_in": 1200,
  "access_token": "<jwt-token>"
}

# Пример неудачного(401) ответа:

{
  "error": "invalid_client",
  "error_description": "Client authentication failed",
  "message": "Client authentication failed"
}
